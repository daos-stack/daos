381 static inline bool
382 daos_label_is_valid(const char *label)
383 {
384  int len;
385  int i;
386  bool maybe_uuid = false;
387 
389  if (label == NULL)
390  return false;
391 
393  len = strnlen(label, DAOS_PROP_LABEL_MAX_LEN + 1);
394  if (len == 0 || len > DAOS_PROP_LABEL_MAX_LEN)
395  return false;
396 
398  for (i = 0; i < len; i++) {
399  char c = label[i];
400 
401  if (isalnum(c) || c == '.' || c == '_' || c == ':')
402  continue;
403  if (c == '-') {
404  maybe_uuid = true;
405  continue;
406  }
407 
408  return false;
409  }
410 
412  if (maybe_uuid && strnlen(label, 36) == 36) {
413  bool is_uuid = true;
414  const char *p;
415 
417  for (i = 0, p = label; i < 36; i++, p++) {
418  if (i == 8 || i == 13 || i == 18 || i == 23) {
419  if (*p != '-') {
420  is_uuid = false;
421  break;
422  }
423  continue;
424  }
425  if (!isxdigit(*p)) {
426  is_uuid = false;
427  break;
428  }
429  }
430 
431  if (is_uuid)
432  return false;
433  }
434 
435  return true;
436 }
