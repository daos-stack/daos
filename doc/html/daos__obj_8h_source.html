193  /*
194  * Type of the value in an iod can be either a single type that is
195  * always overwritten when updated, or it can be an array of EQUAL sized
196  * records where the record is updated atomically. Note that an akey can
197  * only support one type of value which is set on the first update. If
198  * user attempts to mix types in the same akey, the behavior is
199  * undefined, even after the object, key, or value is punched. If
200  * \a iod_type == DAOS_IOD_SINGLE, then iod_nr has to be 1, and
201  * \a iod_size would be the size of the single atomic value. The idx is
202  * ignored and the rx_nr is also required to be 1.
203  */
204  daos_iod_type_t iod_type;
206  daos_size_t iod_size;
207  /*
208  * Number of entries in the \a iod_recxs, \a iod_csums, and \a iod_eprs
