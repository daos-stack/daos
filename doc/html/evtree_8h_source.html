335 static inline char
336 evt_debug_print_visibility(const struct evt_entry *ent)
337 {
338  int flags = EVT_VISIBLE | EVT_PARTIAL | EVT_COVERED;
339 
340  switch (ent->en_visibility & flags) {
341  default:
342  D_ASSERT(0);
343  case 0:
344  break;
345  case EVT_PARTIAL:
346  return 'p';
347  case EVT_VISIBLE:

350  return 'v';
351  case EVT_COVERED:
352  return 'C';
353  case EVT_COVERED | EVT_PARTIAL:
354  return 'c';
355  }
356 
357  return 'U';
358 }
