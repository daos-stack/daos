155 /*
156  * Convert bio_sglist into d_sg_list_t, caller is responsible to
157  * call daos_sgl_fini(sgl, false) to free iovs.
158  */
159 static inline int
160 bio_sgl_convert(struct bio_sglist *bsgl, d_sg_list_t *sgl)
161 {
162  int i, rc;
163 
164  D_ASSERT(sgl != NULL);
165  D_ASSERT(bsgl != NULL);
166 
167  rc = daos_sgl_init(sgl, bsgl->bs_nr_out);
168  if (rc != 0)
169  return -DER_NOMEM;
170 
171  sgl->sg_nr_out = bsgl->bs_nr_out;
172 
173  for (i = 0; i < sgl->sg_nr_out; i++) {
174  struct bio_iov *biov = &bsgl->bs_iovs[i];
175  d_iov_t *iov = &sgl->sg_iovs[i];
176 
177  iov->iov_buf = biov->bi_buf;
178  iov->iov_len = biov->bi_data_len;
179  iov->iov_buf_len = biov->bi_data_len;
180  }
181 
182  return 0;
183 }
